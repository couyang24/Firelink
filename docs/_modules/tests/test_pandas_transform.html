import numpy as np
import pandas as pd

    Fillna,
    Filter,
    Groupby,
    Query,
    Select_dtypes,
)

[docs]def test_query(test_pandas_df):
    expected = pd.DataFrame(
        {
            "a": range(7, 9),
            "b": range(17, 19),
            "c": range(27, 29),
            "d": ["j", "q"],
            "e": ["a", "d"],
        }
    )
    output = (
        Query("a>5 and d in ['j', 'q']")
        .fit_transform(test_pandas_df)
        .reset_index(drop=True)
    )
    assert_frame_equal(output, expected)

[docs]def test_apply(test_pandas_df):

        {
            "index": ["a", "b", "c"],
            0: [45, 145, 245],
        }
    )
    output = (
